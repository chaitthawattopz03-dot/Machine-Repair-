LINE Messaging API — ผ่าน Google Apps Script
วิธีตั้งค่า (ทำครั้งเดียว)
1. ไปที่ script.google.com → เปิด Project ของท่าน
2. ลบ code เดิมออก แล้ววางโค้ด GAS ที่ได้รับด้านล่างนี้ทั้งหมด
3. กด Deploy → New deployment → Web app → Execute as: Me → Who has access: Anyone
4. คัดลอก URL ที่ได้มาวางช่องด้านล่าง
5. ใส่ Group ID (Cxxxxxxx) ที่ได้จาก Webhook → กดบันทึก → ทดสอบ ✅
const CHANNEL_TOKEN = 'วาง_Channel_Access_Token_ที่นี่'; const CLOUDINARY_CLOUD = 'ddizediks'; const CLOUDINARY_PRESET = 'mix_repair'; // อัปโหลดภาพขึ้น Cloudinary — ได้ direct URL ส่ง LINE ได้เลย function uploadToCloudinary(base64Data) { const raw = base64Data.split(',')[1] || base64Data; const res = UrlFetchApp.fetch( 'https://api.cloudinary.com/v1_1/' + CLOUDINARY_CLOUD + '/image/upload', { method: 'post', payload: { file: 'data:image/jpeg;base64,' + raw, upload_preset: CLOUDINARY_PRESET }, muteHttpExceptions: true } ); const json = JSON.parse(res.getContentText()); if (json.secure_url) return json.secure_url; throw new Error('Cloudinary error: ' + JSON.stringify(json)); } function sendToLine(to, messages) { const payload = { to: to, messages: messages }; const options = { method: 'post', contentType: 'application/json', headers: { 'Authorization': 'Bearer ' + CHANNEL_TOKEN }, payload: JSON.stringify(payload), muteHttpExceptions: true }; const res = UrlFetchApp.fetch('https://api.line.me/v2/bot/message/push', options); return JSON.parse(res.getContentText()); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const messages = []; // 1) ข้อความหลัก messages.push({ type: 'text', text: data.message }); // 2) ภาพ — อัปโหลดขึ้น Cloudinary แล้วส่ง URL ให้ LINE if (data.photos && data.photos.length > 0) { data.photos.forEach(function(b64) { try { const imgUrl = uploadToCloudinary(b64); messages.push({ type: 'image', originalContentUrl: imgUrl, previewImageUrl: imgUrl }); Logger.log('uploaded: ' + imgUrl); } catch(imgErr) { Logger.log('img error: ' + imgErr.message); } }); } const result = sendToLine(data.to, messages); Logger.log('LINE result: ' + JSON.stringify(result)); return ContentService .createTextOutput(JSON.stringify( result.message ? {status:'error',error:result.message} : {status:'ok'} )) .setMimeType(ContentService.MimeType.JSON); } catch(err) { Logger.log('doPost error: ' + err.message); return ContentService .createTextOutput(JSON.stringify({status:'error',error:err.message})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService .createTextOutput(JSON.stringify({status:'GAS LINE Proxy + Cloudinary ✅'})) .setMimeType(ContentService.MimeType.JSON); } function testSend() { const groupId = 'C694bb1d6f7e3c66acef88635991467d9'; const result = sendToLine(groupId, [{ type: 'text', text: '🧪 ทดสอบระบบแจ้งซ่อม + Cloudinary ✅' }]); Logger.log(JSON.stringify(result)); }
